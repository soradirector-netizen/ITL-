    render();
}

/* =========================
描画
========================= */
function render() {
    let quizDiv = document.getElementById("quiz");
    quizDiv.innerHTML = "";

    let list = [...mcqQuestions];

    if (randomMode) {
        list.sort(() => Math.random() - 0.5);
    }

    list.forEach(q => {
        let div = document.createElement("div");
        div.className = "card";

        let html = `<h3>Q${q.id}. ${q.q}</h3>`;

        q.choices.forEach((c, i) => {
            html += `
                <label>
                    <input type="radio" name="q${q.id}" value="${i}">
                    ${c}
                </label><br>
            `;
        });

        div.innerHTML = html;
        quizDiv.appendChild(div);
    });

    let essaySection = document.createElement("div");
    essaySection.className = "card";
    essaySection.innerHTML = "<h2>記述問題</h2>";

    essayQuestions.forEach((e, i) => {
        essaySection.innerHTML += `
            <div class="essay">
                <p>Q${i + 1}. ${e.q}</p>
                <textarea name="essay${i}" rows="4" style="width:100%; resize:vertical;"></textarea>
            </div>
        `;
    });

    quizDiv.appendChild(essaySection);
}

/* =========================
記述AI採点（簡易）
========================= */
function gradeEssay() {
    let score = 0;

    essayQuestions.forEach((e, i) => {
        const textarea = document.querySelector(`textarea[name=essay${i}]`);
        const ans = textarea ? textarea.value.trim() : "";

        if (!ans) return;

        let hit = 0;
        e.key.forEach(k => {
            if (ans.includes(k)) hit++;
        });

        if (hit >= 2) score++;
    });

    return score;
}

/* =========================
採点（最終統合版）
========================= */
function checkAnswers() {
    let mcqScore = 0;
    let result = "";

    mcqQuestions.forEach(q => {
        let selected = document.querySelector(`input[name=q${q.id}]:checked`);

        if (!selected) {
            result += `<p class="wrong">Q${q.id}: 未回答</p>`;
            return;
        }

        let val = parseInt(selected.value, 10);

        if (val === q.answer) {
            mcqScore++;
            result += `<p class="correct">Q${q.id}: 正解</p>`;
        } else {
            result += `<p class="wrong">Q${q.id}: 不正解（正解: ${q.choices[q.answer]}）</p>`;
        }
    });

    let essayScore = gradeEssay();
    let resultDiv = document.getElementById("result");

    resultDiv.innerHTML = `
        <div class="card">
            <h2>採点結果</h2>
            <p>選択問題: ${mcqScore} / ${mcqQuestions.length}</p>
            <p>記述問題: ${essayScore} / ${essayQuestions.length}</p>
            <hr>
            ${result}
        </div>
    `;

    resultDiv.scrollIntoView({ behavior: "smooth" });
}
